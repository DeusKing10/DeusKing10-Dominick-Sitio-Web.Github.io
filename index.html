import 'dart:io';

void main() {
  // Precios base para mayores y jóvenes
  double precioMayor = 350;

  // Descuentos por nivel para mayores de 18 años

    1: 0.40,

    3: 0.15,
    4: 0.0, // Sin descuento para otros niveles
  };

  // Descuentos por nivel para jóvenes de 18 años

    1: 0.60,
    2: 0.40,
    3: 0.20,
    4: 0.0, // Sin descuento para otros niveles
  };

  // Solicitar la edad del hombre
  print('Ingresa la edad del hombre:');
  int? edad = int.tryParse(stdin.readLineSync()!);

  if (edad == null || edad < 0) {
    print('Edad inválida.');
    return;
  }

  // Solicitar el nivel de sistema de beneficio
  print('Ingresa el nivel de sistema de beneficio (1, 2, 3, 4):');
  int? nivel = int.tryParse(stdin.readLineSync()!);

  if (nivel == null || !descuentoMayor.containsKey(nivel)) {
    print('Nivel inválido.');
    return;
  }

  double precioFinal = 0;
  double descuentoAplicado = 0;

  // Calcular según la edad
  if (edad > 18) {
    // Hombres mayores de 18 años
    descuentoAplicado = descuentoMayor[nivel]!;
    precioFinal = precioMayor * (1 - descuentoAplicado);
  } else if (edad == 18) {
    // Hombres de 18 años
    descuentoAplicado = descuentoJoven[nivel]!;
    precioFinal = precioJoven * (1 - descuentoAplicado);
  } else {
    print('No se aplica la venta de libreta para menores de 18 años.');
    return;
  }

  // Mostrar resultados
  print('El descuento aplicado es: ${descuentoAplicado * 100}%');
  print('El valor final a pagar es: S/ $precioFinal');
}

            import 'dart:io';

            bool esPrimo(int numero) {
             if (numero < 2) {
              return false;
             }
             for (int i = 2; i <= numero ~/ 2; i++) {
              if (numero % i == 0) {
               return false;
              }
             }
             return true;
            }
            
            int sumarPrimosEnRango(int inicio, int fin) {
             int suma = 0;
             for (int i = inicio; i <= fin; i++) {
              if (esPrimo(i)) {
               suma += i;
              }
             }
             return suma;
            }
            
            void main() {
             stdout.write('Ingresa el número inicial: ');
             int inicio = int.parse(stdin.readLineSync()!);
            
             stdout.write('Ingresa el número final: ');

            // Función para calcular el factorial utilizando BigInt para manejar números grandes
            BigInt calcularFactorial(int n) {
              BigInt resultado = BigInt.one;
              
              for (int i = 1; i <= n; i++) {

            import 'dart:io';

void main() {
  // Solicitar al usuario que ingrese un número entero
  stdout.write("Ingrese un número entero: ");
  int numero = int.parse(stdin.readLineSync()!);

  // Inicializar la variable para almacenar el número invertido
  int invertido = 0;

  // Usar el bucle while para invertir el número
  while (numero != 0) {
    int digito = numero % 10;  // Extraer el último dígito
    invertido = invertido * 10 + digito;  // Agregar el dígito al número invertido
    numero = numero ~/ 10;  // Eliminar el último dígito del número original
  }

  // Mostrar el resultado
  print("El número invertido es: $invertido");
}

                }

              print("Ingrese los valores de la segunda matriz:");
              for (int i = 0; i < n; i++) {

                }
              }
            
              // Sumar las dos matrices
              for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                  sumaMatriz[i][j] = matriz1[i][j] + matriz2[i][j];
                }
              }
            
              // Mostrar el resultado de la suma de las matrices
              print("La suma de las matrices es:");
              for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {

Código N° 07 - EJERCICIO SOBRE MATRÍZ DE ESPIRAL EN EL LENGUAJE DE DART


        void main() {
            int n = 5; // Cambia este valor para probar con diferentes tamaños de matriz
          
            List> matriz = List.generate(n, (_) => List.filled(n, 0));
          
            int valor = 1;
            int filaInicio = 0, filaFin = n - 1;
            int colInicio = 0, colFin = n - 1;
          
            // Bucle para llenar la matriz en espiral
            while (filaInicio <= filaFin && colInicio <= colFin) {
              // Llenar la fila superior (de izquierda a derecha)
              for (int i = colInicio; i <= colFin; i++) {
                matriz[filaInicio][i] = valor++;
              }
              filaInicio++;
          
              // Llenar la columna derecha (de arriba hacia abajo)
              for (int i = filaInicio; i <= filaFin; i++) {
                matriz[i][colFin] = valor++;
              }
              colFin--;
          
              // Llenar la fila inferior (de derecha a izquierda)
              if (filaInicio <= filaFin) {
                for (int i = colFin; i >= colInicio; i--) {
                  matriz[filaFin][i] = valor++;
                }
                filaFin--;
              }
          
              // Llenar la columna izquierda (de abajo hacia arriba)
              if (colInicio <= colFin) {
                for (int i = filaFin; i >= filaInicio; i--) {
                  matriz[i][colInicio] = valor++;
                }
                colInicio++;
              }
            }
          
            // Imprimir la matriz en espiral
            print("Matriz en espiral de tamaño $n:");
            for (var fila in matriz) {
              print(fila);
            }
          }

Código N° 08 - EJERCICIO SOBRE VERIFICACIÓN DE UN NÚMERO ARMSTRONG EN EL LENGUAJE DE DART


        import 'dart:math'; // Para usar pow()
        import 'dart:io';   // Para recibir entrada del usuario
        
        void main() {
          // Solicitar al usuario que ingrese un número
          print("Ingresa un número para verificar si es un número Armstrong:");
          int numero = int.parse(stdin.readLineSync()!);
        
          // Convertir el número en una lista de dígitos
          String numeroStr = numero.toString();
          int n = numeroStr.length; // Número de dígitos
        
          // Inicializar la suma
          int suma = 0;
        
          // Bucle para separar y elevar cada dígito
          for (int i = 0; i < n; i++) {
            int digito = int.parse(numeroStr[i]);
            suma += pow(digito, n).toInt(); // Elevar a la potencia n
          }
        
          // Verificar si el número es Armstrong
          if (suma == numero) {
            print("$numero es un número Armstrong.");
          } else {
            print("$numero no es un número Armstrong.");
          }
        }

PROYECTO N° 01 : - " EJERCICIO CÁLCULO DE DESCUENTOS EN PENSIONES UNIVERSITARIAS EN EL LENGUAJE DE DART " 

  
          /*En una universidad, los alumnos están clasificados en cuatro categorías. 

A cada categoría le corresponde una pensión mensual distinta dada en la siguiente tabla:

Categoría Pensión
A - S/. 550
B - S/. 500
C - S/. 460
D - S/. 400
Semestralmente, la universidad efectúa rebajas en las pensiones de sus estudiantes a partir del 
segundo ciclo basándose en el promedio ponderado del ciclo anterior en porcentajes dados en 
la tabla siguiente:
Promedio Descuento
0 a 13.99 - No hay descuento
14.00 a 15.99 - 10 %
16.00 a 17.99 - 12 %
18.00 a 20.00 - 15 %
Dado el promedio ponderado y la categoría de un estudiante, diseñe un programa que determine 
cuánto de rebaja recibirá sobre su pensión actual y a cuánto asciende su nueva pensión.*/

import 'dart:io';

class Estudiante {
  String categoria;
  double promedio;
  double pension;
  double descuento;
  double nuevaPension;

  Estudiante(this.categoria, this.promedio) {
    pension = obtenerPension(categoria);
    descuento = calcularDescuento(promedio, pension);
    nuevaPension = pension - descuento;
  }

  double obtenerPension(String categoria) {
    switch (categoria) {
      case 'A':
        return 550.0;
      case 'B':
        return 500.0;
      case 'C':
        return 460.0;
      case 'D':
        return 400.0;
      default:
        stdout.write('Categoría no válida. Se asignará categoría D por defecto.\n');
        return 400.0;
    }
  }

  double calcularDescuento(double promedio, double pension) {
    double porcentajeDescuento;

    if (promedio < 14.0) {
      porcentajeDescuento = 0.0;
    } else if (promedio < 16.0) {
      porcentajeDescuento = 0.10; // 10%
    } else if (promedio < 18.0) {
      porcentajeDescuento = 0.12; // 12%
    } else {
      porcentajeDescuento = 0.15; // 15%
    }

    return pension * porcentajeDescuento;
  }

  void mostrarInformacion() {
    print('------------ Reporte del Estudiante ------------');
    print('Categoría: $categoria');
    print('Promedio: $promedio');
    print('Descuento: S/. ${descuento.toStringAsFixed(2)}');
    print('Nueva Pensión: S/. ${nuevaPension.toStringAsFixed(2)}');
    print('------------------------------------------------');
  }
}

void main() {
  List estudiantes = [];

  stdout.write('Ingrese la cantidad de estudiantes: ');

    estudiantes.add(estudiante);

  }
}

PROYECTO N° 02 : - " EJERCICIO EVALUACIÓN DEL DESEMPEÑO DE EMPLEADOS: PUNTUALIDAD Y RENDIMIENTO EN EL LENGUAJE DE DART " 

  
          /*Una empresa evalúa a sus empleados bajo dos criterios: puntualidad y rendimiento. En cada caso, 
el empleado recibe un puntaje que va de 1 a 10, de acuerdo con los siguientes criterios: 

Puntaje por puntualidad: - está en función de los minutos de tardanza de acuerdo con la 
siguiente tabla:
Minutos de tardanza Puntaje
0 - 10
1 a 2 - 8
3 a 5 - 6
6 a 9 - 4
Mas de - 9 0
Puntaje por rendimiento: - está en función de la cantidad de observaciones efectuadas al 
empleado por no cumplir sus obligaciones de acuerdo con la siguiente tabla:
Observaciones efectuadas Puntaje
0 - 10
1 - 8
2 - 5
3 - 1
Mas de 3 - 0
El puntaje total del empleado es la suma del puntaje por puntualidad más el puntaje por 
rendimiento. Basándose en el puntaje total, el empleado recibe una bonificación anual de acuerdo 
con la siguiente tabla:
Puntaje total Bonificación
Menos de 11 - S/. 2.5 por punto
11 a 13 - S/. 5.0 por punto
14 a 16 - S/. 7.5 por punto
17 a 19 - S/. 10.0 por punto
20 - S/. 12.5 por punto
Dados los minutos de tardanza y el número de observaciones de un empleado, diseñe un 
programa que determine el puntaje por puntualidad, el puntaje por rendimiento, el puntaje total 
y la bonificación que le corresponden.*/

import 'dart:io';

class Empleado {
  String nombre;
  int minutosTardanza;
  int observaciones;

  Empleado(this.nombre, this.minutosTardanza, this.observaciones);

  // Método para calcular puntaje por puntualidad
  int calcularPuntajePuntualidad() {
    if (minutosTardanza == 0) return 10;
    if (minutosTardanza >= 1 && minutosTardanza <= 2) return 8;
    if (minutosTardanza >= 3 && minutosTardanza <= 5) return 6;
    if (minutosTardanza >= 6 && minutosTardanza <= 9) return 4;
    return 0; // Más de 9 minutos
  }

  // Método para calcular puntaje por rendimiento
  int calcularPuntajeRendimiento() {
    if (observaciones == 0) return 10;
    if (observaciones == 1) return 8;
    if (observaciones == 2) return 5;
    if (observaciones == 3) return 1;
    return 0; // Más de 3 observaciones
  }

  // Método para calcular puntaje total
  int calcularPuntajeTotal() {
    return calcularPuntajePuntualidad() + calcularPuntajeRendimiento();
  }

  // Método para calcular la bonificación
  double calcularBonificacion() {
    int puntajeTotal = calcularPuntajeTotal();
    if (puntajeTotal < 11) return puntajeTotal * 2.5;
    if (puntajeTotal >= 11 && puntajeTotal <= 13) return puntajeTotal * 5.0;
    if (puntajeTotal >= 14 && puntajeTotal <= 16) return puntajeTotal * 7.5;
    if (puntajeTotal >= 17 && puntajeTotal <= 19) return puntajeTotal * 10.0;
    return puntajeTotal * 12.5; // Puntaje total de 20
  }

  // Método para mostrar el reporte del empleado
  void mostrarReporte() {
    int puntajePuntualidad = calcularPuntajePuntualidad();
    int puntajeRendimiento = calcularPuntajeRendimiento();
    int puntajeTotal = calcularPuntajeTotal();
    double bonificacion = calcularBonificacion();

    print('------------ Reporte de $nombre ------------');
    print('Puntaje por Puntualidad: $puntajePuntualidad');
    print('Puntaje por Rendimiento: $puntajeRendimiento');
    print('Puntaje Total: $puntajeTotal');
    print('Bonificación: S/. ${bonificacion.toStringAsFixed(2)}');
    print('--------------------------------------------');
  }
}

void main() {
  List empleados = [];

  stdout.write('¿Cuántos empleados desea evaluar?: ');
  int cantidad = int.parse(stdin.readLineSync()!);

  for (int i = 0; i < cantidad; i++) {
    stdout.write('Ingrese el nombre del empleado ${i + 1}: ');
    String nombre = stdin.readLineSync()!;

    stdout.write('Ingrese los minutos de tardanza de $nombre: ');
    int minutosTardanza = int.parse(stdin.readLineSync()!);

    stdout.write('Ingrese la cantidad de observaciones de $nombre: ');
    int observaciones = int.parse(stdin.readLineSync()!);

    // Crear un empleado y agregarlo a la lista
    Empleado empleado = Empleado(nombre, minutosTardanza, observaciones);
    empleados.add(empleado);
  }

  // Mostrar el reporte de todos los empleados
  for (Empleado empleado in empleados) {
    empleado.mostrarReporte();
  }
}

          /*Una tienda vende tres tipos de productos a los precios dados en la siguiente tabla: 

Producto Precio

Como oferta la tienda ofrece un regalo de acuerdo con la siguiente tabla:
Unidades adquiridas Regalo
